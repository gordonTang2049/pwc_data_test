Road Accident Analysis¶

prepared by Gordon Tang¶

Purpose of this road accident analysis is to understand the data of road accidents and assist of allocating resource to reduce road accidents.

Top 10 Highest Number of Road Accidents by Region¶

In [4]:
import seaborn as sns
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
acc_infor = pd.read_csv('ACCIDENT.csv')
acc_loc = pd.read_csv('ACCIDENT_LOCATION.csv')
node = pd.read_csv('NODE.csv')
mer_acc_infor_node = acc_infor.merge(node, on='ACCIDENT_NO', how='left')
series = mer_acc_infor_node[['LGA_NAME']]\
    .groupby(['LGA_NAME']).LGA_NAME.count().sort_values(ascending=False)

seriesh = series.head(10)
areas = seriesh.keys()

sns.set_style('darkgrid')
x = areas
y = seriesh.values
plt.figure(figsize=(16, 6))
sns.barplot(x, y)\
    .set_title("Top 10 of highest Road Accidents Region")
plt.xlabel("Name of Region", size=10)
plt.ylabel("Number of Accident", size=10)
print(seriesh)
plt.show()
